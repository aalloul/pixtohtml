>>> rt = spidermonkey.Runtime()
>>> cx = rt.new_context()
>>> cx.add_global("Monkey", Monkey)
>>> monkey = cx.execute('var x = new Monkey(); x.baz = "schmammo"; x;')
>>> monkey.baz
u'schmammo'
>>> monkey.__class__.__name__
'Monkey'
